# Trapezoid Method

\# Define function to integrate
def f(x):
    return 1/(1 + x**2)
\# Implementing trapezoid method
def trapezoid(x0,xn,n):
    \# calculating step size
    h = (xn - x0) / n
    \# Finding sum 
    integration = f(x0) + f(xn)
    for i in range(1,n):
        k = x0 + i*h
        integration = integration + 2 * f(k)
    \# Finding final integration value
    integration = integration * h/2
    return integration
\# Input section

sub_interval = int(input("Masukkan jumlah sub interval: "))
\# Call trapezoid() method and get result
result = trapezoid(lower_limit, upper_limit, sub_interval)
print("Hasil integrasi dengan metode Trapesium adalah: %0.6f" % (result) )
